function setupRealtime() {
  const channel = supabase
    .channel("siao-realtime")
    .on("postgres_changes", { event: "*", schema: "public", table: "structures" }, async () => {

async function loginSIAO() {
  const pwd = document.getElementById("siao-pwd").value;
  if (pwd === SIAO_PWD) {
    currentUser = { role: "siao" };
    await loadData();
    showScreen("screen-admin");
    initAdmin();
    setupRealtime();
  } else {
    showError("Mot de passe incorrect");
  }
}

async function loginStructure() {
  const id = document.getElementById("struct-select").value;
  const pwd = document.getElementById("struct-pwd").value.trim();
  if (!id) { showError("Sélectionnez une structure"); return; }
  const s = structures.find(x => x.id === parseInt(id));
  if (s && pwd.toUpperCase() === s.code.toUpperCase()) {
    currentUser = { role: "structure", id: s.id, nom: s.nom };
    await loadData();
    showScreen("screen-structure");
    initStructure(s);
    setupRealtime();
  } else {
    showError("Code incorrect");
  }
}

function showError(msg) {
  const el = document.getElementById("login-error");
  el.textContent = msg;
  el.classList.add("show");
  setTimeout(() => el.classList.remove("show"), 3000);
}

function adminTab(tab, btn) {
  document.querySelectorAll("#screen-admin .tab-panel").forEach(p => p.classList.remove("active"));
  document.querySelectorAll("#screen-admin .sb-btn").forEach(b => b.classList.remove("active"));
  document.getElementById("admin-" + tab).classList.add("active");
  if (btn) btn.classList.add("active");
  if (tab === "vue-ensemble") renderVueEnsemble();
  if (tab === "agenda") renderAgenda();
  if (tab === "creneaux") renderCreneaux();
  if (tab === "structures") renderStructures();
  if (tab === "reservations") renderReservations();
}

function renderVueEnsemble() {
  const libres = creneaux.filter(c => c.statut === "libre").length;
  const reserves = creneaux.filter(c => c.statut === "reserve").length;
  document.getElementById("admin-stats").innerHTML = `
    <div class="stat-card" style="--c:var(--blue)"><div class="stat-val">${creneaux.length}</div><div class="stat-lbl">Total créneaux</div></div>
    <div class="stat-card" style="--c:var(--green)"><div class="stat-val">${libres}</div><div class="stat-lbl">Disponibles</div></div>
    <div class="stat-card" style="--c:var(--orange)"><div class="stat-val">${reserves}</div><div class="stat-lbl">Réservés</div></div>
    <div class="stat-card" style="--c:var(--accent)"><div class="stat-val">${structures.length}</div><div class="stat-lbl">Structures</div></div>
  `;
  const next = creneaux.filter(c => c.statut === "libre").sort((a,b) => new Date(a.date+" "+a.heure) - new Date(b.date+" "+b.heure)).slice(0,4);
  document.getElementById("admin-next-creneaux").innerHTML = next.length ? next.map(c => `<div style="padding:8px 0;border-bottom:1px solid var(--border)"><strong>${c.type}</strong><br>${c.date} ${c.heure}</div>`).join("") : "<div>Aucun créneau</div>";
  const recent = creneaux.filter(c => c.statut === "reserve").slice(0,4);
  document.getElementById("admin-recent-resa").innerHTML = recent.length ? recent.map(c => `<div style="padding:8px 0;border-bottom:1px solid var(--border)">${c.type}<br>${c.date}</div>`).join("") : "<div>Aucune réservation</div>";
}
